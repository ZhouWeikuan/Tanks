160 {

163  {
164  //podajemy początkową dowolną pozycję, bo nie znamy wymiarów pocisku

167 
168  Direction tmp_d = (testFlag(TSF_ON_ICE) ? new_direction : direction);
169  switch(tmp_d)

171  case D_UP:
172  bullet->pos_x += (dest_rect.w - bullet->dest_rect.w) / 2;
173  bullet->pos_y -= bullet->dest_rect.h - 4;
174  break;
175  case D_RIGHT:

178  break;
179  case D_DOWN:

181  bullet->pos_y += dest_rect.h - 4;

183  case D_LEFT:

186  break;
187  }
188

191  bullet->speed = AppConfig::bullet_default_speed * 1.3;

240  m_slip_time = AppConfig::slip_time;
241  }
242  else

244 
245  if(!stop)
246  {

248  int pos_x_tile, pos_y_tile;
249  switch (direction)
250  {

253  pos_x_tile = ((int)(pos_x / AppConfig::tile_rect.w)) * AppConfig::tile_rect.w;
254  if(pos_x - pos_x_tile < epsilon) pos_x = pos_x_tile;
255  else if(pos_x_tile + AppConfig::tile_rect.w - pos_x < epsilon) pos_x = pos_x_tile + AppConfig::tile_rect.w;
256  break;
257  case D_RIGHT:
258  case D_LEFT:
259  pos_y_tile = ((int)(pos_y / AppConfig::tile_rect.h)) * AppConfig::tile_rect.h;
260  if(pos_y - pos_y_tile < epsilon) pos_y = pos_y_tile;
261  else if(pos_y_tile + AppConfig::tile_rect.h - pos_y < epsilon) pos_y = pos_y_tile + AppConfig::tile_rect.h;
262  break;
263  }
