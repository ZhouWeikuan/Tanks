385 void Game::loadLevel(std::string path)
386 {
387  std::fstream level(path, std::ios::in);
388  std::string line;
389  int j = -1;
390 
391  if(level.is_open())
392  {
393  while(!level.eof())
394  {
395  std::getline(level, line);
396  std::vector<Object*> row;
397  j++;
398  for(unsigned i = 0; i < line.size(); i++)
399  {
400  Object* obj;
401  switch(line.at(i))
402  {
403  case '#' : obj = new Brick(i * AppConfig::tile_rect.w, j * AppConfig::tile_rect.h); break;
404  case '@' : obj = new Object(i * AppConfig::tile_rect.w, j * AppConfig::tile_rect.h, ST_STONE_WALL); break;
405  case '%' : m_bushes.push_back(new Object(i * AppConfig::tile_rect.w, j * AppConfig::tile_rect.h, ST_BUSH)); obj = nullptr; break;
406  case '~' : obj = new Object(i * AppConfig::tile_rect.w, j * AppConfig::tile_rect.h, ST_WATER); break;
407  case '-' : obj = new Object(i * AppConfig::tile_rect.w, j * AppConfig::tile_rect.h, ST_ICE); break;
408  default: obj = nullptr;
409  }
410  row.push_back(obj);
411  }
412  m_level.push_back(row);
413  }
414  }
415 
416  m_level_rows_count = m_level.size();
417  if(m_level_rows_count)
418  m_level_columns_count = m_level.at(0).size();
419  else m_level_columns_count = 0;
420

424  //wyczyszczenie miejsca orzełeka
425  for(int i = 12; i < 14 && i < m_level_columns_count; i++)
426  {
427  for(int j = m_level_rows_count - 2; j < m_level_rows_count; j++)
428  {
429  if(m_level.at(j).at(i) != nullptr)

480 void Game::checkCollisionTankWithLevel(Tank* tank, Uint32 dt)
481 {

491  switch(tank->direction)
492  {

497  column_end = (tank->collision_rect.x + tank->collision_rect.w) / AppConfig::tile_rect.w + 1;
498  break;

501  column_end = column_start + 1;
502  row_start = tank->collision_rect.y / AppConfig::tile_rect.h - 1;
503  row_end = (tank->collision_rect.y + tank->collision_rect.h) / AppConfig::tile_rect.h + 1;

509  column_end = (tank->collision_rect.x + tank->collision_rect.w) / AppConfig::tile_rect.w + 1;
510  break;
511  case D_LEFT:
512  column_end = tank->collision_rect.x / AppConfig::tile_rect.w;
513  column_start = column_end - 1;

528  for(int j = column_start; j <= column_end ;j++)

534

536

538  if(intersect_rect.w > 0 && intersect_rect.h > 0)

610 void Game::checkCollisionBulletWithLevel(Bullet* bullet)
611 {
612  if(bullet == nullptr) return;

614 
615  int row_start, row_end;
616  int column_start, column_end;
617 
618  SDL_Rect* br, *lr;
619  SDL_Rect intersect_rect;
620  Object* o;
621 
622  //========================kolizja z elementami mapy========================
623  switch(bullet->direction)
624  {
625  case D_UP:
626  row_start = row_end = bullet->collision_rect.y / AppConfig::tile_rect.h;
627  column_start = bullet->collision_rect.x / AppConfig::tile_rect.w;
628  column_end = (bullet->collision_rect.x + bullet->collision_rect.w) / AppConfig::tile_rect.w;
629  break;
630  case D_RIGHT:
631  column_start = column_end = (bullet->collision_rect.x + bullet->collision_rect.w) / AppConfig::tile_rect.w;
632  row_start = bullet->collision_rect.y / AppConfig::tile_rect.h;
633  row_end = (bullet->collision_rect.y + bullet->collision_rect.h) / AppConfig::tile_rect.h;
634  break;
635  case D_DOWN:
636  row_start = row_end = (bullet->collision_rect.y + bullet->collision_rect.h)/ AppConfig::tile_rect.h;
637  column_start = bullet->collision_rect.x / AppConfig::tile_rect.w;
638  column_end = (bullet->collision_rect.x + bullet->collision_rect.w) / AppConfig::tile_rect.w;
639  break;
640  case D_LEFT:
641  column_start = column_end = bullet->collision_rect.x / AppConfig::tile_rect.w;
642  row_start = bullet->collision_rect.y / AppConfig::tile_rect.h;
643  row_end = (bullet->collision_rect.y + bullet->collision_rect.h) / AppConfig::tile_rect.h;
644  break;
645  }
646  if(column_start < 0) column_start = 0;
647  if(row_start < 0) row_start = 0;
648  if(column_end >= m_level_columns_count) column_end = m_level_columns_count - 1;
649  if(row_end >= m_level_rows_count) row_end = m_level_rows_count - 1;
650 
651  br = &bullet->collision_rect;
652 
653  for(int i = row_start; i <= row_end; i++)
654  for(int j = column_start; j <= column_end; j++)

658  if(o->type == ST_ICE || o->type == ST_WATER) continue;
659 
660  lr = &o->collision_rect;
661  intersect_rect = intersectRect(lr, br);
662 
663  if(intersect_rect.w > 0 && intersect_rect.h > 0)
664  {
665  if(bullet->increased_damage)

667  delete o;
668  m_level.at(i).at(j) = nullptr;
669  }
670  else if(o->type == ST_BRICK_WALL)

674  if(brick->to_erase)

678  }
679  }
680  bullet->destroy();
681  }
682  }
